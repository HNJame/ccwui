说明文档

基于 jQuery 开发的表单验证组件。配合jquery.tips.js组件，提供更友好的交互。

使用方法与参数说明： 

    $( "selector" ).validate({
        fields: {
            "txt-required": {					//需要验证字段的id
                required: true					//该字段不能为空
            },
            "txt-max-length": {		
                maxLength: 5					//该字段最多字符数
            },
            "txt-digits-max": {		
                digits: true,					//该字段只能填写数字
                max: 100						//数字最大值
            },
            "txt-email": {		
                email: true						//Email验证
            },
            "txt-url": {		
                url: true						//URL验证
            },
            "txt-password": {		
                required: true,		
                minLength: 6					//该字段最少6位
            },
            "txt-confirm-password": {		
                equalTo: "txt-password",		//密码验证，两次输入密码是否一致
                msg: "两次输入的密码不一致"		//自定义错误提示信息
            },
            "radio-status": {		
                required: true,					//单选按钮验证
                position: "top"					//错误信息显示位置，默认 right。只有引入jquery.tips.js有效。
            },
            "cb-group": {		
                required: true,					//多选按钮验证
                position: "top"
            }
        },
        error: function( e, field ) {
            var tipEl = field.el;
            
            if ( tipEl.is( "input:checkbox, input:radio" )) {
                tipEl = tipEl.parents( ".field-input" );
            }
            
            tipEl.tips({
                content: field.msg,
                position: "right",
                color: "red",
                autoHide: false
            });
        },
        valid: function( e, field ) {
            field.el.tips( "hide" );
        }
    });

表单验证API: 

	$( ".form" ).validate( "validate" );	//传递字符串 "validate" 验证所有字段是否满足上面验证条件
	$( ".form" ).validate( "reset" )		//传递字符串 "reset" 清除所有错误信息提示
